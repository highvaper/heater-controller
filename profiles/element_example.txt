# Element Heater Profile Example
# This profile is configured for a resistive heating element (nichrome wire)
# When loaded, missing values will use defaults from shared_state

# ===== TEMPERATURE & CONTROL =====
# Temperature unit: string ('C' or 'F')
temperature_units=C

# Target temperature setpoint: int (degrees)
temperature_setpoint=180

# Maximum allowed setpoint: int (degrees) - protects PTFE around thermocouple
temperature_max_allowed_setpoint=250

# Control mode: string ('temperature_pid', 'watts', or 'duty_cycle')
control=temperature_pid

# ===== HEATER CONFIGURATION =====
# Heater type: string ('element' or 'induction')
# - 'element' uses resistive heating element with PWM control (default)
# - 'induction' uses induction coils with coil switching
heater_type=element

# Heater element resistance in ohms: float
# Common nichrome wire resistances:
# - 0.4-0.6 ohms for heavier gauge wire
# - 0.8-1.2 ohms for lighter gauge wire
heater_resistance=0.49

# Maximum watts limit: int (watts)
# Safety limit to prevent element from overheating/glowing
# Recommended: 75-100W for most setups
# Note: 14 AWG copper wire max ~15A at 12V = 180W theoretical max
max_watts=75

# ===== PID TUNING =====
# PID tuning parameters: float,float,float (P,I,D values)
# These control how the PID responds to temperature changes
# Element heaters typically need different tuning than induction
pid_temperature_tunings=2.3,0.03,0

# Degrees above setpoint at which PID will be reset if temperature spikes after reaching setpoint
pid_reset_high_temperature=8

# ===== POWER CONFIGURATION =====
# Power type: string ('mains', 'lipo', or 'lead')
power_type=mains

# LIPO battery configuration (when power_type='lipo')
lipo_count=4
lipo_safe_volts=3.3

# Safe minimum voltage for lead acid batteries: float (volts)
lead_safe_volts=12.0

# Safe maximum voltage for mains: float (volts)
mains_safe_volts=28.0

# ===== SESSION TIMING =====
# Session timeout in seconds: int
session_timeout=420

# Session extend time in seconds: int
session_extend_time=120

# ===== AUTOSESSION =====
# Enable autosession data logging: boolean (true or false)
autosession_logging_enabled=false

# Default autosession profile to load (leave empty or omit to use current_autosession_profile.txt)
# default_autosession_profile=

# ===== DISPLAY =====
# Display contrast level: int (0-255)
display_contrast=255

# Rotate display: boolean (true or false)
display_rotate=true
