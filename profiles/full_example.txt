# Comprehensive Profile - All Configuration Options
# This file shows all available settings that can be loaded into SharedState
# Missing values will use defaults from shared_state

# ===== TEMPERATURE & CONTROL =====
# Temperature unit: string ('C' or 'F')
temperature_units=C

# Target temperature setpoint: int (degrees)
temperature_setpoint=165

# Maximum allowed setpoint: int (degrees) - protects PTFE around thermocouple
temperature_max_allowed_setpoint=250

# Control mode: string ('temperature_pid', 'watts', or 'duty_cycle')
# - 'temperature_pid' uses PID controller with temperature setpoint
# - 'watts' uses fixed wattage (0-max_watts)
# - 'duty_cycle' uses direct duty cycle control (0-100%)
control=temperature_pid

# Fixed wattage level when control='watts': int (watts)
set_watts=30

# Duty cycle percentage when control='duty_cycle': float (0.0-100.0)
# Supports decimal values (e.g., 12.3 for 12.3%). 0.1% resolution is supported.
set_duty_cycle=10.0

# ===== HEATER CONFIGURATION =====
# Heater coil/element resistance in ohms: float
heater_resitance=0.49

# Maximum watts limit: int (watts)
# Safety limit to prevent coil from overheating/glowing
max_watts=75

# Temperature difference threshold for induction heaters: int (degrees)
# Used to filter readings when induction heater is on
heater_on_temperature_difference_threshold=20

# ===== POWER THRESHOLDS =====
# Power threshold for control adjustments: int
power_threshold=0

# ===== POWER CONFIGURATION =====
# Power type: string ('mains', 'lipo', or 'lead')
# - 'mains': AC/DC mains supply
# - 'lipo': Lithium polymer batteries
# - 'lead': Lead acid batteries
power_type=mains

# LIPO battery configuration (when power_type='lipo')
# Number of LIPO cells in series: int
lipo_count=4

# Safe minimum voltage per LIPO cell: float (volts)
lipo_safe_volts=3.3

# Safe minimum voltage for lead acid batteries: float (volts)
lead_safe_volts=12.0

# Safe maximum voltage for mains: float (volts)
mains_safe_volts=28.0

# ===== TIMING =====
# Session timeout in seconds: int
# Will be converted to milliseconds internally
session_timeout=420

# Session extend time in seconds: int
# Duration to extend session when user clicks in last minute
session_extend_time=120

# Click check timeout in milliseconds: int
# Window for detecting multi-click inputs
click_check_timeout=800

# Reset PID stats when near setpoint: boolean (true or false)
# Helps reduce overshoot by resetting PID statistics
session_reset_pid_when_near_setpoint=true

# ===== TEMPERATURE MONITORING =====
# PI Pico temperature limit: int (degrees C)
# System shuts down heater if PI temperature exceeds this
pi_temperature_limit=60

# ===== PID TUNING =====
# PID tuning parameters: float,float,float (P,I,D values)
# These control how the PID responds to temperature changes
pid_tunings=2.3,0.03,0

# Degrees above setpoint at which PID will be reset if temperature spikes after reaching setpoint
pid_reset_high_temperature=15

# ===== DISPLAY =====
# Display contrast level: int (0-255)
display_contrast=255

# Rotate display: boolean (true or false)
display_rotate=true
